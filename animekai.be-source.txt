    // ── Copier: copy data-attr value (or href) to clipboard ─────────
    $(document).on('click', '.copier', function (e) {
        e.preventDefault();
        var $el   = $(this);
        var attr  = $el.data('attr');
        var text  = attr ? $el.attr(attr) : $el.text().trim();
        var tip   = bootstrap.Tooltip.getInstance($el[0]);
        navigator.clipboard.writeText(text).then(function () {
            if (tip) {
                tip.setContent({ '.tooltip-inner': 'Copied!' });
                setTimeout(function () { tip.setContent({ '.tooltip-inner': 'Copy' }); }, 1500);
            }
            showToast('Copied to clipboard!', 'success');
        }).catch(function () {
            var ta = document.createElement('textarea');
            ta.value = text; document.body.appendChild(ta); ta.select();
            document.execCommand('copy'); document.body.removeChild(ta);
            showToast('Copied!', 'success');
        });
    });
});
